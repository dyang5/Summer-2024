# filler c function
c := proc(F::Array, a, b)
	lenF := LinearAlgebra[Dimension](F);
	for i from 1 to lenF - 1 do
		if b >= F[i] and b <= F[i+1] then
			return qe_pf(a, F[i]);
		end if;
	end do;
end proc;

usp0 := proc(n)
  local res, i, j;
  res := 1;
  for i from 1 to n do
      res := res * sum( c(i-j,i)*(-cat(y,n+1-i))^j, j=0..i );
  end do;

  return res;
end proc;
